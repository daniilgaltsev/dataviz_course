import warnings

warnings.filterwarnings("ignore")
import altair as alt
import pandas as pd
import numpy as np
np.random.seed(1103)
x = np.random.normal(size=(100,))
y = x * 2.3 - 0.2 + np.random.normal(size=(100,))
x.shape, y.shape
((100,), (100,))
df = pd.DataFrame(dict(x=x, y=y))

alt.Chart(df).mark_circle().encode(
    x="x",
    y="y"
).interactive()
df["feat1"] = np.random.randint(0, 3, 100)
df["feat1"] = df["feat1"].astype(str)
